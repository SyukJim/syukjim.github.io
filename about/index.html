Code for Youtube downloader
This is the example code without using Tkinter. This was used to test the downloading functionality of Pytube. 

from pytube import YouTube

def download_video(url):
    yt = YouTube(url)
    
    # Check if 1080p resolution is available
    if "1080p" in yt.streams.filter(progressive=True).order_by("resolution").desc().first().resolution:
        stream = yt.streams.filter(progressive=True, resolution="1080p").first()
    else:
        stream = yt.streams.get_highest_resolution()
    
    # Download the video
    stream.download(output_path="save path here")

Code below will use Tkinter to download Youtube video. 

import tkinter as tk
from pytube import YouTube
from tkinter import ttk
from PIL import Image, ImageTk
from io import BytesIO
import requests

def show_video_details():
    video_url = entry.get()
    try:
        yt = YouTube(video_url)
        title_label.config(text="Title: " + yt.title)
        
        thumbnail_url = yt.thumbnail_url
        response = requests.get(thumbnail_url)
        image = Image.open(BytesIO(response.content))
        image = image.resize((200, 150))
        thumbnail = ImageTk.PhotoImage(image)
        thumbnail_label.config(image=thumbnail)
        thumbnail_label.image = thumbnail

    except Exception as e:
        title_label.config(text="Error: " + str(e))
        thumbnail_label.config(image="")

def download_video():
    video_url = entry.get()
    resolution = resolution_var.get()
    try:
        yt = YouTube(video_url)
        title = yt.title
        streams = yt.streams.filter(res=resolution, progressive=True, mime_type='video/mp4')
        if streams:
            stream = streams.first()
            filename = title + '.' + stream.subtype
            stream.download(output_path="download path here", filename=filename)
            status_label.config(text="Download completed!")
        else:
            status_label.config(text="No suitable video stream found for the selected resolution.")
    except Exception as e:
        status_label.config(text="Error occurred: " + str(e))

def clear_entry():
    entry.delete(0, tk.END)
    title_label.config(text="")
    thumbnail_label.config(image="")

def paste_from_clipboard():
    entry.delete(0, tk.END)
    entry.insert(tk.END, window.clipboard_get())

# Create the main window
window = tk.Tk()
window.title("YouTube Video Downloader")

# Create labels and entry field
label = tk.Label(window, text="Enter YouTube URL:")
label.grid(row=0, column=0, padx=5, pady=5)
entry = tk.Entry(window, width=50)
entry.grid(row=0, column=1, padx=5, pady=5)

# Create paste button
paste_button = tk.Button(window, text="Paste", command=paste_from_clipboard)
paste_button.grid(row=0, column=2, padx=5, pady=5)

# Create clear button
clear_button = tk.Button(window, text="Clear", command=clear_entry)
clear_button.grid(row=0, column=3, padx=5, pady=5)

# Create resolution dropdown menu
resolutions = ["1080p", "720p", "480p", "360p", "240p", "144p"]
resolution_var = tk.StringVar(window)
resolution_var.set(resolutions[0])  # Default resolution
resolution_dropdown = tk.OptionMenu(window, resolution_var, *resolutions)
resolution_dropdown.grid(row=1, column=2, columnspan=2, padx=5, pady=5)

# Create show details button
details_button = tk.Button(window, text="Show Details", command=show_video_details)
details_button.grid(row=1, column=0, padx=0, pady=5)

# Create download button
download_button = tk.Button(window, text="Download", command=download_video)
download_button.grid(row=1, column=1, padx=0, pady=5)


# Create title label
title_label = tk.Label(window, text="")
title_label.grid(row=3, column=0, columnspan=3, padx=5, pady=5)

# Create thumbnail label
thumbnail_label = tk.Label(window)
thumbnail_label.grid(row=4, column=0, columnspan=3, padx=5, pady=5)

# Create progress bar
progress_bar = ttk.Progressbar(window, orient="horizontal", length=300, mode="determinate")
progress_bar.grid(row=5, column=0, columnspan=3, padx=5, pady=5)

# Create status label
status_label = tk.Label(window, text="")
status_label.grid(row=6, column=0, columnspan=3, padx=5, pady=5)

# Start the main event loop
window.mainloop()
